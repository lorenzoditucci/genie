90 std::string bitsettostring(std::bitset<bitset_size> b, const uint16_t readlen,
91  const encoder_global_b<bitset_size> &egb) {
92  // destroys bitset b
93  static const char revinttochar[8] = {'A', 'N', 'G', 0, 'C', 0, 'T', 0};
94  std::string s;
95  s.resize(readlen);
96  unsigned long long ull;
97  for (int i = 0; i < 3 * readlen / 63 + 1; i++) {
98  ull = (b & egb.mask63).to_ullong();
99  b >>= 63;
100  for (int j = 21 * i; j < 21 * i + 21 && j < readlen; j++) {
101  s[j] = revinttochar[ull % 8];
102  ull /= 8;
103  }
104  }
105  return s;
106 }
