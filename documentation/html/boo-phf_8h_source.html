153 inline unsigned int popcount_32(unsigned int x) {
154  unsigned int m1 = 0x55555555;
155  unsigned int m2 = 0x33333333;
156  unsigned int m4 = 0x0f0f0f0f;
157  unsigned int h01 = 0x01010101;
158  x -= (x >> 1) & m1; /* put count of each 2 bits into those 2 bits */
159  x = (x & m2) + ((x >> 2) & m2); /* put count of each 4 bits in */
160  x = (x + (x >> 4)) & m4; /* put count of each 8 bits in partie droite 4bit piece*/
161  return (x * h01) >> 24; /* returns left 8 bits of x + (x<<8) + ... */
162 }
