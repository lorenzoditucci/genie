538  updaterefcount<bitset_size>(read[current], ref, revref, count, false, true, shift,
539  read_lengths[current], ref_len, rg);
540  if (!left_search) {
541  cur_read_pos = ref_pos + ref_len_old + shift - read_lengths[current];
542  ref_pos = ref_pos + ref_len_old + shift - ref_len;
543  } else {
544  cur_read_pos = ref_pos - shift;
545  ref_pos = cur_read_pos;
546  }
547  if (prev_unmatched == true) // prev read not singleton, write it now
548  {
549  foutRC << 'd';
550  foutorder.write((char *)&prev, sizeof(uint32_t));
551  foutflag << 0; // for unmatched
552  int64_t zero = 0;
553  foutpos.write((char *)&zero, sizeof(int64_t));
554  foutlength.write((char *)&read_lengths[prev], sizeof(uint16_t));
555  }
556  foutRC << (left_search ? 'd' : 'r');
557  foutorder.write((char *)&current, sizeof(uint32_t));
558  foutflag << 1; // for matched
559  foutpos.write((char *)&cur_read_pos, sizeof(int64_t));
560  foutlength.write((char *)&read_lengths[current], sizeof(uint16_t));
561 
562  prev_unmatched = false;
563  break;
564  }
565 
566  revref <<= 2;
567  ref >>= 2;
568  }
569  if (flag == 0) // no match found
570  {
571  num_unmatched_past_1M_thr++;
572  if (!left_search) // start left search
573  {
574  left_search = true;
575  left_search_start = true;
576  // update ref and count with RC of first_read
577  updaterefcount<bitset_size>(read[first_rid], ref, revref, count, true, true, 0,
578  read_lengths[first_rid], ref_len, rg);
579  ref_pos = 0;
580  cur_read_pos = 0;
581  } else // left search done, now pick arbitrary read and start new
582  // contig
583  {
584  left_search = false;
585  for (int64_t j = remainingpos; j >= 0; j--) {
586  if (remainingreads[j] == 1) {
587 #ifdef GENIE_USE_OPENMP
588  read_lock[reorder_lock_idx(j)].set();
589 #endif
590  if (remainingreads[j]) // checking again inside critical block
591  {
592  current = j;
593  remainingpos = j - 1;
594  remainingreads[j] = 0;
595  flag = 1;
596  unmatched[tid]++;
597  }
598 #ifdef GENIE_USE_OPENMP
599  read_lock[reorder_lock_idx(j)].unset();
600 #endif
601  if (flag == 1) break;
602  }
603  }
604  if (flag == 0) {
605  if (prev_unmatched == true) // last read was singleton, write it now
606  {
607  foutorder_s.write((char *)&prev, sizeof(uint32_t));
608  }
609  done = 1; // no reads left
610  } else {
611  updaterefcount<bitset_size>(read[current], ref, revref, count, true, false, 0,
612  read_lengths[current], ref_len, rg);
613  ref_pos = 0;
614  cur_read_pos = 0;
615  if (prev_unmatched == true) // prev read singleton, write it now
616  {
617  foutorder_s.write((char *)&prev, sizeof(uint32_t));
618  }
619  prev_unmatched = true;
620  first_rid = current;
621  prev = current;
622  }
623  }
624  }
625  } // while (!done) end
626 
627  foutRC.close();
628  foutorder.close();
629  foutflag.close();
630  foutpos.close();
631  foutorder_s.close();
632  foutlength.close();
633  for (int i = 0; i < 4; i++) delete[] count[i];
634  delete[] count;
635  } // parallel end
636 
637  delete[] remainingreads;
638 #ifdef GENIE_USE_OPENMP
639  delete[] dict_lock;
640  delete[] read_lock;
641 #endif
642  std::cout << "Reordering done, " << std::accumulate(unmatched, unmatched + rg.num_thr, 0) << " were unmatched\n";
643  for (int i = 0; i < rg.max_readlen; i++) delete[] mask[i];
644  delete[] mask;
645  delete[] mask1;
646  delete[] unmatched;
647  return;
648 }

653 template <size_t bitset_size>
654 void writetofile(std::bitset<bitset_size> *read, uint16_t *read_lengths, reorder_global<bitset_size> &rg) {
655  //
656  // This loop must be executed in parallel with the same #threads
657  // that was used for the hot loop in reorder(), for correctness.
658  // It also shows up in the execution profile, and is worth
659  // parallelizing anyway.
660  //
661  // FIXME re-code the loop so that is doesn't necessarily need to
662  // execute on the same #threads.
663  //
664 #ifdef GENIE_USE_OPENMP
665 #pragma omp parallel num_threads(rg.num_thr)
666 #endif
667  {
668 #ifdef GENIE_USE_OPENMP
669  uint32_t tid = omp_get_thread_num();
670 #else
671  uint32_t tid = 0; // set thread ID to zero if not using OpenMP
672 #endif
673  std::string tid_str = std::to_string(tid);
674  std::ofstream fout(rg.outfile + '.' + tid_str, std::ofstream::out);
675  std::ofstream fout_s(rg.outfile + ".singleton." + tid_str, std::ofstream::out);
676  std::ifstream finRC(rg.outfileRC + '.' + tid_str, std::ifstream::in);
677  std::ifstream finorder(rg.outfileorder + '.' + tid_str, std::ifstream::in | std::ios::binary);
678  std::ifstream finorder_s(rg.outfileorder + ".singleton." + tid_str, std::ifstream::in | std::ios::binary);
679  char s[MAX_READ_LEN + 1], s1[MAX_READ_LEN + 1];
680  uint32_t current;
681  char c;
682  while (finRC >> std::noskipws >> c) // read character by character
683  {
684  finorder.read((char *)&current, sizeof(uint32_t));
685  bitsettostring<bitset_size>(read[current], s, read_lengths[current], rg);
686  if (c == 'd') {
687  fout << s << "\n";
688  } else {
689  reverse_complement(s, s1, read_lengths[current]);
690  fout << s1 << "\n";
691  }
692  }
693  finorder_s.read((char *)&current, sizeof(uint32_t));
694  while (!finorder_s.eof()) {
695  bitsettostring<bitset_size>(read[current], s, read_lengths[current], rg);
696  fout_s << s << "\n";
697  finorder_s.read((char *)&current, sizeof(uint32_t));
698  }
699 
700  fout.close();
701  fout_s.close();
702  finRC.close();
703  finorder.close();
704  finorder_s.close();
705  }
706 
707  // Now combine the num_thr order files
708  std::ofstream fout_s(rg.outfile + ".singleton", std::ofstream::out);
709  std::ofstream foutorder_s(rg.outfileorder + ".singleton", std::ofstream::out | std::ios::binary);
710  for (int tid = 0; tid < rg.num_thr; tid++) {
711  std::string tid_str = std::to_string(tid);
712  std::ifstream fin_s(rg.outfile + ".singleton." + tid_str, std::ifstream::in);
713  std::ifstream finorder_s(rg.outfileorder + ".singleton." + tid_str, std::ifstream::in | std::ios::binary);
714 
715  fout_s << fin_s.rdbuf(); // write entire file
716  foutorder_s << finorder_s.rdbuf();
717 
718  // clear error flags which occur on rdbuffing empty file
719  fout_s.clear();
720  foutorder_s.clear();
721 
722  fin_s.close();
723  finorder_s.close();
724 
725  remove((rg.outfile + ".singleton." + tid_str).c_str());
726  remove((rg.outfileorder + ".singleton." + tid_str).c_str());
727  }
728  fout_s.close();
729  foutorder_s.close();
730  return;
731 }
